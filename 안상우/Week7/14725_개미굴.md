```python
import sys
import heapq

class Node:
    def __init__(self, val = None):
        self.val = val
        self.children = {}
        self.order = []

class Trie:
    def __init__(self):
        self.root = Node()
    
    def insert(self, path: list):
        cur = self.root
        for food in path:
            if food not in cur.children:
                cur.children[food] = Node(food)
                heapq.heappush(cur.order, food)
            cur = cur.children[food]
    
    def printTrie(self, cur: Node, d):
        depth = d
        if depth > 0:
            print("--" * (depth - 1) + cur.val)
        while cur.order:
            food = heapq.heappop(cur.order)
            self.printTrie(cur.children[food], depth + 1)

N = int(sys.stdin.readline())
foodTrie = Trie()
for _ in range(N):
    path = sys.stdin.readline().split()
    del path[0]
    foodTrie.insert(path)
foodTrie.printTrie(foodTrie.root, 0)
```
