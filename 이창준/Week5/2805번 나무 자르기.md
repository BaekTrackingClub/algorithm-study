import sys
import heapq

input = sys.stdin.read
lines = input().splitlines()

N, M = map(int, lines[0].split()) # N: 나무의 수, M: 상근이가 희망하는 나무의 길이

Trees = list(map(int, lines[1].split()))
Trees.sort()
min_height = float('inf')

def binary_search(arr, M):
    start , end = 0, max(arr)
    result = 0
    while start <= end:
        mid = (start + end)//2
        woods_sum = 0
        
        for tree in arr:
            if tree > mid:
                woods_sum += tree - mid
        
        if woods_sum >= M:
            result = mid
            start = mid + 1
            
        else:
            end = mid - 1
    return result

print(binary_search(Trees, M))
